Capstone: Build a Personal Research Agent

Module 1: Foundations of Agentic AI Duration: ~2 hours Prerequisite: Labs 1, 2, and 3

Your Mission

Build an agent that takes a research topic, autonomously investigates it using multiple tools, and produces a structured research report — with zero human intervention after the initial prompt.

Requirements Recap

Pass (70+): ✅ Accepts a research topic from the user ☐ Generates at least 3 research questions ☐ Uses at least 2 tools to gather information ☐ Produces a structured report (Summary, Key Findings, Sources) ☐ Saves the report to a file Distinction (80+): All above, plus: ☐ Confidence levels per finding (high/medium/low) ☐ Gap analysis and follow-up questions ☐ 3+ different tools ☐ Graceful error handling Honors (90+): All above, plus: ☐ Persistent memory across sessions ☐ Multiple output formats (Markdown + JSON) ☐ Self-evaluation section

def search_knowledge(query: str) -> str:
    """Simulate a knowledge search using a separate Claude call.
    
    In production, this would be a real search API (Google, Bing, Tavily, etc.).
    For this capstone, we use Claude as a knowledge retrieval system.
    
    Args:
        query: The search query.
    
    Returns:
        Relevant information about the query.
    """
    search_response = client.messages.create(
        model=MODEL,
        max_tokens=1024,
        system=(
            "You are a research database. When given a query, return factual, "
            "detailed information with specific data points, dates, and statistics "
            "where available. Format as a list of key facts. Be comprehensive but "
            "concise. If you're uncertain about a fact, say so."
        ),
        messages=[{"role": "user", "content": f"Search query: {query}"}]
    )
    return search_response.content[0].text

def run_research_agent(
    topic: str,
    max_iterations: int = 20,
    verbose: bool = True,
) -> str:
    """Run the research agent on a given topic.
    
    Args:
        topic: The research topic to investigate.
        max_iterations: Maximum tool-use cycles.
        verbose: Print step-by-step reasoning.
    
    Returns:
        The agent's final response.
    """
    # ──────────────────────────────────────────────────────────
    # TODO: Write the system prompt for your research agent.
    # This is where you define WHO the agent is and HOW it works.
    # 
    # Your system prompt should instruct the agent to:
    # 1. Generate research questions about the topic
    # 2. Use search_knowledge to gather information for each question
    # 3. Synthesize findings into a structured report
    # 4. Save the report to a file
    # 5. (Distinction) Rate confidence per finding
    # 6. (Distinction) Identify gaps and suggest follow-ups
    # 7. (Honors) Save key findings to memory
    # ──────────────────────────────────────────────────────────
    
    system_prompt = """
    TODO: Write your research agent's system prompt here.
    This is the most important part of the capstone.
    Refer to Lab 1 (system prompts) for guidance.
    """
    
    user_message = f"Research the following topic and produce a comprehensive report: {topic}"
    messages = [{"role": "user", "content": user_message}]
    iteration = 0
    tools_used = []  # Track which tools were used
    
    if verbose:
        print(f"\n{'='*60}")
        print(f"RESEARCH TOPIC: {topic}")
        print(f"{'='*60}")
    
    while iteration < max_iterations:
        iteration += 1
        
        if verbose:
            print(f"\n── Step {iteration} ──")
        
        response = client.messages.create(
            model=MODEL,
            max_tokens=4096,
            system=system_prompt,
            tools=tools,
            messages=messages
        )
        
        # Check if agent is done
        if response.stop_reason == "end_turn":
            final = ""
            for block in response.content:
                if hasattr(block, "text"):
                    final += block.text
            
            if verbose:
                print(f"\n✓ Research complete in {iteration} steps")
                print(f"  Tools used: {tools_used}")
            
            return final
        
        # Process tool calls
        tool_results = []
        for block in response.content:
            if hasattr(block, "text") and block.text:
                if verbose:
                    print(f"  💭 {block.text[:200]}")
            
            if block.type == "tool_use":
                name = block.name
                inp = block.input
                tools_used.append(name)
                
                if verbose:
                    print(f"  🔧 {name}({json.dumps(inp)[:100]})")
                
                func = TOOL_FUNCTIONS.get(name)
                if func:
                    try:
                        result = func(**inp)
                    except Exception as e:
                        result = f"Error: {e}"
                else:
                    result = f"Error: Unknown tool '{name}'"
                
                if verbose:
                    display = result[:150] + "..." if len(str(result)) > 150 else result
                    print(f"  📋 {display}")
                
                tool_results.append({
                    "type": "tool_result",
                    "tool_use_id": block.id,
                    "content": str(result)
                })
        
        messages.append({"role": "assistant", "content": response.content})
        messages.append({"role": "user", "content": tool_results})
    
    return "Error: Research agent hit iteration limit."

print("✓ Research agent ready")
